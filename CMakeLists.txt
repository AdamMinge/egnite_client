# -------------------------------------------------------------------------------------------------- #
# -------------------------------- Set CMake version and project ----------------------------------- #
# -------------------------------------------------------------------------------------------------- #
cmake_minimum_required(VERSION 3.11)
project(EGNITE VERSION 0.1.0 LANGUAGES CXX)
# -------------------------------------------------------------------------------------------------- #
# ---------------------------------- Include config and macros ------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
include(${EGNITE_SOURCE_DIR}/cmake/config.cmake)
include(${EGNITE_SOURCE_DIR}/cmake/macros.cmake)
# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Add subdirectories ---------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
add_subdirectory(src/egnite)

if(EGNITE_BUILD_TEST)
    add_subdirectory(tests)
endif()
# -------------------------------------------------------------------------------------------------- #
# ---------------------------------------- Export targets ------------------------------------------ #
# -------------------------------------------------------------------------------------------------- #
egnite_export_modules()
# -------------------------------------------------------------------------------------------------- #
# ------------------------------ Generate and install documentation -------------------------------- #
# -------------------------------------------------------------------------------------------------- #
if(EGNITE_BUILD_DOCUMENTATION)
    egnite_generate_documentation()
    egnite_install_documentation()
endif()
# -------------------------------------------------------------------------------------------------- #
